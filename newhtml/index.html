    from flask import Flask
# 匯入 render_template 以使用樣板
from flask import render_template

app = Flask(__name__)

@app.route('/')
def index():
    return 'Hello, from Flask!'

# route 名稱必須以斜線開頭
@app.route('/home')
def home():
    return render_template('home.html')

app.run(host='0.0.0.0', port=81)
